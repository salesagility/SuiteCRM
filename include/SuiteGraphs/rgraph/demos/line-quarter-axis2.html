A quarter axis Line chart

[No canvas support]

This goes in the documents header: 
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.drawing.xaxis.js"></script>
<script src="RGraph.drawing.yaxis.js"></script>
<script src="RGraph.line.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250">
    [No canvas support]
</canvas>
This is the code that generates the chart: 
<script>
    window.onload = function ()
    {
        var data = [48,46,45,-5,-4,4,3];
        var co = document.getElementById("cvs").getContext('2d');
        
        // Draw the background grid manually
        co.beginPath();
        co.strokeStyle = '#ddd';
            
            // Draw the vertical lines
            for (var i=35; i<=575; i+=20) co.rect(i + 0.5, 25, 0, 300);
            
            // Draw the horizontal lines
            for (var i=25; i<=325; i+=25) co.rect(35, i + 0.5, 540, 0);
        co.stroke();


        // Draw the top Y axis
        var top_yaxis = new RGraph.Drawing.YAxis({
            id: 'cvs',
            x: 35,
            options: {
                max: 50,
                numticks: 0,
                numlabels: 5,
                gutterTop: 25,
                gutterBottom: 75,
                textSize:14
            }
        }).draw()



        // Draw the bottom Y axis
        var bottom_yaxis = new RGraph.Drawing.YAxis({
            id: 'cvs',
            x: 35,
            options: {
                max: 10,
                gutterTop: 275,
                gutterBottom: 25,
                numlabels: 1,
                numticks: 0,
                noendtickTop: true,
                scaleInvert: true,
                scaleZerostart: false,
                unitsPre: '-',
                textSize:14
            }
        }).draw()

        // Draw the center X axis (which has no labels)
        var xaxis = new RGraph.Drawing.XAxis({
            id: 'cvs',
            y: 275,
            options: {
                labels: [],
                numticks: 0,
                hmargin: 25,
                gutterLeft: 35
            }
        }).draw()

        // Draw the bottom X axis (which is transparent but does have labels)
        var xaxis = new RGraph.Drawing.XAxis({
            id: 'cvs',
            y: 325,
            options: {
                xaxispos: 'center',
                gutterLeft: 35,
                labels: ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'],
                labelsPosition: 'edge',
                numticks: 0,
                hmargin: 25,
                colors: ['rgba(0,0,0,0)'],
                textColor: '#666',
                textSize: 14
            }
        }).draw()


        
        // Draw the line chart
        var line = new RGraph.Line({
            id: 'cvs',
            data: data,
            options: {
                hmargin: 25,
                gutterBottom: 75,
                gutterLeft: 35,
                outofbounds: true,
                linewidth: 1,
                ylabels: false,
                noaxes: true,
                backgroundGrid: false,
                colors: ['gray'],
                ymax: 50,
                textSize: 14
            }
        }).draw();
    };
</script>
