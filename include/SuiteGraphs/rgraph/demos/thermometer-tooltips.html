A thermometer chart with onmousemove driven tooltips

[No canvas support]

This goes in the documents header: 
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.tooltips.js"></script>
<script src="RGraph.thermometer.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="100" height="400">
    [No canvas support]
</canvas>
This is the code that generates the chart: 
<script>
    window.onload = function ()
    {
        var thermometer = new RGraph.Thermometer({
            id: 'cvs',
            min: 0,
            max: 100,
            value: 68,
            options: {
                tooltips: ["It's getting <b style='color: red'>hot</b> in here!"],
                tooltipsEvent: 'mousemove'
            }
        }).draw();
    };
</script>
